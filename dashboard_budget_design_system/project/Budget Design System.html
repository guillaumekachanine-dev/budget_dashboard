16 — Récapitulatif CSS Variables
:root {
  /* Primary */
  --primary-500: #5B57F5;   /* ★ couleur principale */
  --primary-600: #4642D9;   /* hover states */
  --primary-700: #332FBD;   /* gradient end */
  --primary-50:  #EEEEFF;   /* backgrounds subtils */

  /* Sémantiques */
  --color-positive: #2ED47A;
  --color-negative: #FC5A5A;
  --color-warning:  #FFAB2E;
  --color-info:     #4B8BF4;

  /* Neutrals */
  --neutral-0:   #FFFFFF;   /* card bg */
  --neutral-50:  #F8F9FE;   /* page bg */
  --neutral-100: #EBEBF5;   /* tracks, séparateurs */
  --neutral-400: #9898B3;   /* labels secondaires */
  --neutral-700: #2E2E50;   /* texte */
  --neutral-800: #1C1C3A;   /* texte principal */
  --neutral-900: #0D0D1F;   /* heading */

  /* Typographie */
  --font-sans: 'DM Sans', sans-serif;
  --font-mono: 'DM Mono', monospace;

  /* Radius */
  --radius-sm:   8px;    /* icônes, badges */
  --radius-md:   12px;   /* items, petites cartes */
  --radius-lg:   16px;   /* cartes moyennes */
  --radius-xl:   20px;   /* cartes */
  --radius-2xl:  24px;   /* grandes cartes, screens */
  --radius-full: 9999px; /* pills, avatars, FAB */

  /* Ombres */
  --shadow-card: 0 2px 12px rgba(28,28,58,0.07);
  --shadow-md:   0 4px 16px rgba(91,87,245,0.12);
  --shadow-lg:   0 8px 32px rgba(91,87,245,0.16);
}
